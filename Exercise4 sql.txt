Write a query to get unique department ID from employee table.

ans--select distinct department_id from employees;

***
Write a query to get all employee details from the employee table order by first name, descending.

ans--select * from employees order by first_name desc;

***
Write a query to get the employee ID, names (first_name, last_name), salary in ascending order of
salary.

ans--select e.employee_id,e.first_name,e.last_name,e.salary from employees e order by salary asc;

Display details of employee with ID 150 or 160.

ans--select * from employees where employee_id=150 or employee_id=160;

***
